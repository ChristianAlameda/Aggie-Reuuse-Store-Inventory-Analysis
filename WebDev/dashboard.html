function uploadImage() { var fileInput = document.createElement("input"); fileInput.type = "file"; fileInput.accept = "image/*"; fileInput.addEventListener("change", function(event) { var file = event.target.files[0]; if (file) { alert("File uploaded: " + file.name); } else { alert("Please select a file to upload."); } }); fileInput.click(); }
